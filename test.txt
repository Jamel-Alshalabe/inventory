function handleLogin(){
  const u=document.getElementById('login-user').value.trim();
  const p=document.getElementById('login-pass').value.trim();
  const err=document.getElementById('login-error');
  if(!u||!p){err.textContent='الرجاء إدخال بيانات الدخول';err.classList.remove('hidden');return}

  // Check built-in users first
  let found=builtinUsers.find(x=>x.username===u&&x.password===p);
  if(!found){
    // Then check SDK users
    const sysUsers=getUsers();
    const sysUser=sysUsers.find(x=>x.username===u&&x.password_hash===p);
    if(sysUser){
      found={username:sysUser.username,role:sysUser.role,warehouse:sysUser.assigned_warehouse};
    }
  }
  if(!found){err.textContent='بيانات الدخول غير صحيحة';err.classList.remove('hidden');return}

  currentUser=found;
  err.classList.add('hidden');
  if(currentUser.role==='user'&&currentUser.warehouse!=='all'){
    currentWarehouse=currentUser.warehouse;
  }
  document.getElementById('login-screen').classList.add('hidden');
  document.getElementById('main-app').classList.remove('hidden');
  document.getElementById('main-app').classList.add('flex');
  setupAfterLogin();
}

function renderDashboardChart(movs){
  const canvas=document.getElementById('dashboard-chart');

  const ctx=canvas.getContext('2d');
  
  // Last 7 days data
  const days=7;
  const labels=[];
  const inData=[];
  const outData=[];
  
  for(let i=days-1;i>=0;i--){
    const d=new Date();
    d.setDate(d.getDate()-i);
    const dateStr=d.toISOString().split('T')[0];
    labels.push(d.toLocaleDateString('ar-EG',{month:'short',day:'numeric'}));
    
    const dayMovs=movs.filter(m=>{
      const mDate=(m.date||'').split('T')[0];

  // Clear previous chart
  if(window.dashboardChart){
    window.dashboardChart.destroy();
  }
  
  // Create chart - ensure proper initialization
  try{
    window.dashboardChart=new Chart(ctx,{
      type:'line',
      data:{
        labels:labels,
        datasets:[
          {label:'وارد',data:inData,borderColor:'#22c55e',backgroundColor:'rgba(34,197,94,0.1)',borderWidth:2,fill:true,tension:0.4},

        ]
      },
      options:{
        responsive:true,
        maintainAspectRatio:true,
        plugins:{legend:{labels:{color:'#e0e0e0'}}},
        scales:{
          y:{ticks:{color:'#a0a0a0'},grid:{color:'#333'}},
          x:{ticks:{color:'#a0a0a0'},grid:{color:'#333'}}
        }
      }
    });
  }catch(e){console.error('Chart error:',e)}

// Excel Import
function handleExcelUpload(event){
  const file=event.target.files[0];
  if(!file){showToast('الرجاء اختيار ملف','error');return}
  
  const validTypes=['.xlsx','.xls','.csv','application/vnd.openxmlformats-officedocument.spreadsheetml.sheet','application/vnd.ms-excel','text/csv'];
  const fileName=file.name.toLowerCase();
  const isValid=validTypes.some(t=>fileName.endsWith(t)||file.type.includes(t.replace(/^\./, '')));
  
  if(!isValid){
    showToast('⚠️ صيغة الملف غير صحيحة. استخدم .xlsx أو .xls أو .csv','error');
    document.getElementById('excel-file-input').value='';
    return;
  }
  
  const reader=new FileReader();
  reader.onload=function(e){
    try{
      let workbook,rows;
      
      if(fileName.endsWith('.csv')){
        // Parse CSV
        const text=new TextDecoder().decode(e.target.result);
        const lines=text.split('\n');
        rows=lines.map(line=>line.split(',').map(cell=>cell.trim().replace(/^"|"$/g,'')));
      }else{
        // Parse Excel
        const data=new Uint8Array(e.target.result);
        workbook=XLSX.read(data,{type:'array',raw:false});
        const worksheet=workbook.Sheets[workbook.SheetNames[0]];
        if(!worksheet){
          showToast('⚠️ الملف لا يحتوي على بيانات','error');
          document.getElementById('excel-file-input').value='';
          return;
        }
        rows=XLSX.utils.sheet_to_json(worksheet,{header:1,defval:''});
      }
      
      if(!rows||rows.length<2){
        showToast('⚠️ الملف فارغ أو لا يحتوي على صفوف بيانات','error');
        document.getElementById('excel-file-input').value='';
        return;
      }
      
      // Parse products from rows
      importedProducts=[];
      const existingCodes=getProducts().map(p=>p.code);
      
      // Skip header row and process data
      for(let i=1;i<rows.length;i++){
        const row=rows[i];
        if(!row||row.length<5)continue; // Skip rows with insufficient columns
        
        // Clean and parse each cell
        const code=(row[0]||'').toString().trim();
        const name=(row[1]||'').toString().trim();
        const qtyRaw=(row[2]||'').toString().trim().replace(/,/g,'.');
        const buyRaw=(row[3]||'').toString().trim().replace(/,/g,'.');
        const sellRaw=(row[4]||'').toString().trim().replace(/,/g,'.');
        
        const qty=parseFloat(qtyRaw)||0;
        const buyPrice=parseFloat(buyRaw)||0;
        const sellPrice=parseFloat(sellRaw)||0;
        
        // Validate
        if(!code||!name||code.toLowerCase()==='كود'||name.toLowerCase()==='اسم'){
          continue; // Skip header rows
        }
        
        if(qty<=0||buyPrice<0||sellPrice<0){
          importedProducts.push({code,name,qty,buyPrice:buyRaw,sellPrice:sellRaw,status:'❌ قيم غير صحيحة'});
          continue;
        }
        
        if(!code||!name){
          importedProducts.push({code,name,qty,buyPrice,sellPrice,status:'❌ بيانات ناقصة'});
          continue;
        }
        
        if(existingCodes.includes(code)){
          importedProducts.push({code,name,qty,buyPrice,sellPrice,status:'⚠️ موجود مسبقاً'});
          continue;
        }
        
        importedProducts.push({code,name,qty,buyPrice,sellPrice,status:'✅ جاهز'});
      }
      
      if(importedProducts.length===0){
        showToast('⚠️ لم يتم العثور على بيانات صحيحة في الملف','error');
        document.getElementById('excel-file-input').value='';
        return;
      }
      
      // Show preview
      renderImportPreview();
      showToast('✓ تم قراءة الملف بنجاح - '+importedProducts.length+' منتج');
    }catch(err){
      console.error('Excel parsing error:',err);
      showToast('⚠️ خطأ في قراءة الملف:\n'+err.message.substring(0,50),'error');
      document.getElementById('excel-file-input').value='';
    }
  };
  
  reader.onerror=function(err){
    console.error('File read error:',err);
    showToast('⚠️ خطأ في فتح الملف','error');
    document.getElementById('excel-file-input').value='';
  };
  
  reader.readAsArrayBuffer(file);
}

    }

async function saveMovement(){
  const type=document.getElementById('m-type').value;
  const pCode=document.getElementById('m-product').value;
  const qty=parseInt(document.getElementById('m-qty').value)||0;
  let price=parseFloat(document.getElementById('m-price').value);
  const date=document.getElementById('m-date').value||new Date().toISOString().split('T')[0];

  if(!pCode||qty<=0){showToast('الرجاء اختيار المنتج وإدخال الكمية','error');return}
  if(allData.length>=999){showToast('تم الوصول للحد الأقصى','error');return}

  // If price not entered, use product default price
  const prod=getProducts().find(p=>p.code===pCode);
  if(!prod){showToast('المنتج غير موجود','error');return}
  
  if(!price || price === 0){
    price=type==='in'?prod.buy_price:prod.sell_price;
  }
  
  if(type==='out'&&(prod.quantity||0)<qty){showToast('الكمية غير كافية في المخزون (متوفر: '+prod.quantity+')','error');return}
  
  const newQty=type==='in'?(prod.quantity||0)+qty:(prod.quantity||0)-qty;
  await window.dataSdk.update({...prod,quantity:newQty});

  const r=await window.dataSdk.create({type:'movement',movement_type:type,product_code:pCode,movement_qty:qty,movement_price:price,warehouse:currentWarehouse,date:new Date(date).toISOString()});
  if(r.isOk){
    showToast('تم تسجيل الحركة');
    hideModal('movement-modal');
    await addLog(type==='in'?'وارد':'صادر',`${pCode} × ${qty}`);
    // Reset form fields completely
    document.getElementById('m-qty').value='';
    document.getElementById('m-price').value='';
    document.getElementById('m-date').value=new Date().toISOString().split('T')[0];
    document.getElementById('m-type').value='in';
    document.getElementById('m-product').value='';
    // Reload products in dropdown
    const sel=document.getElementById('m-product');
    if(sel){
      const products=getProducts();

async function deleteMovement(id){
  requestDelete(async()=>{
    const rec=allData.find(d=>d.__backendId===id);
    if(!rec)return;
    // Reverse quantity
    const prod=getProducts().find(p=>p.code===rec.product_code);
    if(prod){
      const newQty=rec.movement_type==='in'?(prod.quantity||0)-(rec.movement_qty||0):(prod.quantity||0)+(rec.movement_qty||0);
      await window.dataSdk.update({...prod,quantity:Math.max(0,newQty)});
    }
    const r=await window.dataSdk.delete(rec);
    if(r.isOk){showToast('تم حذف الحركة');addLog('حذف حركة',rec.product_code)}
    else showToast('حدث خطأ','error');
  });
}

function addInvoiceItem(){
  const prods=getProducts();
  const list=document.getElementById('invoice-items-list');
  const idx=invoiceItemCount++;
  const div=document.createElement('div');
  div.className='grid grid-cols-12 gap-2 items-end';
  div.id='inv-item-'+idx;
  div.dataset.index=idx;
  
  // Generate options - ensure at least one option exists
  const productOptions=prods.length>0?prods.map(p=>`<option value="${p.code}" data-price="${p.sell_price||0}">${p.name} (${p.code})</option>`).join(''):'<option value="">لا توجد منتجات متاحة</option>';

${productOptions}
</select></div>
<div class="col-span-3"><input id="ii-qty-${idx}" type="number" min="1" value="1" onchange="updateInvoiceTotal()" class="w-full bg-dark-800 border border-dark-500 rounded-lg px-2 py-2 text-white text-xs focus:border-brand-500 focus:outline-none" placeholder="كمية"></div>
<div class="col-span-3"><input id="ii-price-${idx}" type="number" min="0" step="0.01" value="0" onchange="updateInvoiceTotal()" class="w-full bg-dark-800 border border-dark-500 rounded-lg px-2 py-2 text-white text-xs focus:border-brand-500 focus:outline-none" placeholder="سعر"></div>
<div class="col-span-1"><button onclick="document.getElementById('inv-item-${idx}').remove();updateInvoiceTotal()" class="p-2 text-red-400 hover:text-red-300"><i data-lucide="x" class="w-3 h-3"></i></button></div>`;
  list.appendChild(div);

  // Set default price and event listeners
  const selEl=document.getElementById('ii-prod-'+idx);
  const priceEl=document.getElementById('ii-price-'+idx);
  
  if(selEl){
    // Set initial price from selected product
    if(selEl.selectedOptions[0]&&selEl.selectedOptions[0].dataset.price){
      priceEl.value=selEl.selectedOptions[0].dataset.price;
    }
    
    // Update price when product changes
    selEl.addEventListener('change',function(){
      const opt=this.selectedOptions[0];
      if(opt&&opt.dataset.price){
        priceEl.value=opt.dataset.price;
      }else{
        priceEl.value=0;
      }
      updateInvoiceTotal();
    });
  }
  
  lucide.createIcons();
  updateInvoiceTotal();
}

function updateInvoiceTotal(){
  let total=0;
  for(let i=0;i<invoiceItemCount;i++){
    const el=document.getElementById('inv-item-'+i);
    if(!el)continue;
    const q=parseFloat(document.getElementById('ii-qty-'+i)?.value)||0;
    const p=parseFloat(document.getElementById('ii-price-'+i)?.value)||0;
    total+=q*p;
  }
  document.getElementById('inv-total').textContent=fmt(total)+' '+cs();
}

function getInvoiceItems(){
  const items=[];
  for(let i=0;i<invoiceItemCount;i++){
    const el=document.getElementById('inv-item-'+i);
    if(!el)continue;
    const code=document.getElementById('ii-prod-'+i)?.value;
    const qty=parseFloat(document.getElementById('ii-qty-'+i)?.value)||0;
    const price=parseFloat(document.getElementById('ii-price-'+i)?.value)||0;
    if(code&&qty>0)items.push({code,qty,price});
  }
  return items;
}

async function saveInvoice(){
  const customer=document.getElementById('inv-customer').value.trim();
  const items=getInvoiceItems();
  if(!customer||items.length===0){showToast('الرجاء إدخال اسم العميل والمنتجات','error');return}
  if(allData.length>=999){showToast('تم الوصول للحد الأقصى','error');return}

  const prods=getProducts();
  // Verify all items exist and have sufficient quantity
  for(const item of items){
    const prod=prods.find(p=>p.code===item.code);
    if(!prod){showToast(`المنتج ${item.code} غير موجود`,'error');return}
    if((prod.quantity||0)<item.qty){showToast(`الكمية غير كافية: ${prod.name} (متوفر: ${prod.quantity})`,'error');return}
  }

  const total=items.reduce((s,it)=>s+it.qty*it.price,0);
  const invNum='INV-'+Date.now().toString().slice(-6);

  // Update product quantities (sell)
  for(const item of items){
    const prod=prods.find(p=>p.code===item.code);
    if(prod){
      await window.dataSdk.update({...prod,quantity:(prod.quantity||0)-item.qty});
      // Record movement
      if(allData.length<999){
        await window.dataSdk.create({type:'movement',movement_type:'out',product_code:item.code,movement_qty:item.qty,movement_price:item.price,warehouse:currentWarehouse,date:new Date().toISOString()});
      }
    }
  }

  const r=await window.dataSdk.create({type:'invoice',invoice_number:invNum,customer_name:customer,invoice_items:JSON.stringify(items),invoice_total:total,invoice_date:new Date().toISOString(),warehouse:currentWarehouse});
  if(r.isOk){
    showToast('تم إنشاء الفاتورة');
    hideModal('invoice-modal');
    addLog('فاتورة جديدة',`${invNum} - ${customer}`);
    // Reset form completely
    document.getElementById('inv-customer').value='';
    invoiceItemCount=0;
    document.getElementById('invoice-items-list').innerHTML='';
    addInvoiceItem();
    renderInvoices();
  }else showToast('حدث خطأ','error');
}

// Reports
function genReport(period){
  const now=new Date();
  let from=new Date(0);
  if(period==='week'){from=new Date(now);from.setDate(from.getDate()-7)}
  else if(period==='month'){from=new Date(now);from.setMonth(from.getMonth()-1)}
  else if(period==='year'){from=new Date(now);from.setFullYear(from.getFullYear()-1)}

  document.getElementById('report-from').value=from.toISOString().split('T')[0];
  document.getElementById('report-to').value=now.toISOString().split('T')[0];
  renderReport(from,now);
}

function renderReport(from,to){
  const movs=getMovements().filter(m=>{const d=new Date(m.date||0);return d>=from&&d<=to});
  const prods=getProducts();
  const totalIn=movs.filter(m=>m.movement_type==='in').reduce((s,m)=>s+(m.movement_qty||0)*(m.movement_price||0),0);
  const totalOut=movs.filter(m=>m.movement_type==='out').reduce((s,m)=>s+(m.movement_qty||0)*(m.movement_price||0),0);
  const remaining=prods.reduce((s,p)=>s+(p.quantity||0),0);

  document.getElementById('rpt-in').textContent=fmt(totalIn)+' '+cs();
  document.getElementById('rpt-out').textContent=fmt(totalOut)+' '+cs();
  document.getElementById('rpt-remaining').textContent=fmt(remaining);
  document.getElementById('rpt-profit').textContent=fmt(totalOut-totalIn)+' '+cs();

  const tb=document.getElementById('report-table');
  if(!tb)return;
  const sorted=movs.sort((a,b)=>new Date(b.date||0)-new Date(a.date||0));
  tb.innerHTML=sorted.map(m=>{
    const prod=prods.find(p=>p.code===m.product_code);
    const total=(m.movement_qty||0)*(m.movement_price||0);
    return`<tr class="border-t border-dark-600"><td class="px-4 py-2 text-dark-200 text-xs">${m.date?new Date(m.date).toLocaleDateString('ar-EG'):'-'}</td>

function downloadFile(filename,content,type='text/plain'){
  try{
    // Ensure content is string
    let contentStr=typeof content==='string'?content:JSON.stringify(content);
    
    // Add BOM for UTF-8 encoding if CSV
    if(type.includes('csv')){
      if(!contentStr.startsWith('\uFEFF')){
        contentStr='\uFEFF'+contentStr;
      }
    }
    
    // Create Uint8Array for proper encoding
    const encoder=new TextEncoder();
    const uint8Array=encoder.encode(contentStr);
    const blob=new Blob([uint8Array],{type:type+';charset=utf-8'});
    
    // Create download link
    const url=URL.createObjectURL(blob);
    const link=document.createElement('a');
    link.href=url;
    link.download=filename;
    link.style.display='none';
    link.style.visibility='hidden';
    
    // Add to DOM
    document.body.appendChild(link);
    
    // Trigger download
    link.dispatchEvent(new MouseEvent('click',{bubbles:true,cancelable:true,view:window}));
    
    // Cleanup after delay
    setTimeout(()=>{
      try{
        if(document.body.contains(link)){
          document.body.removeChild(link);
        }
        URL.revokeObjectURL(url);
      }catch(e){
        console.error('Cleanup error:',e);
      }
    },500);
    
    showToast(`✅ تم تحميل: ${filename}`);
  }catch(e){
    console.error('Download error:',e);
    showToast('⚠️ حدث خطأ أثناء التحميل','error');
  }
}

  const movs=getMovements().filter(m=>{
    if(m.product_code!==prod.code)return false;
    const d=new Date(m.date||0);
    const f=new Date(from);
    const t=new Date(to);
    return d>=f&&d<=t;
  });
  
  const totalIn=movs.filter(m=>m.movement_type==='in').reduce((s,m)=>s+(m.movement_qty||0),0);
  const totalOut=movs.filter(m=>m.movement_type==='out').reduce((s,m)=>s+(m.movement_qty||0),0);
  const totalInValue=movs.filter(m=>m.movement_type==='in').reduce((s,m)=>s+(m.movement_qty||0)*(m.movement_price||0),0);

function exportReportExcel(){
  const from=document.getElementById('report-from').value;
  const to=document.getElementById('report-to').value;
  const movs=getMovements().filter(m=>{const d=new Date(m.date||0);const f=new Date(from);const t=new Date(to);return d>=f&&d<=t});
  const prods=getProducts();
  
  const totalIn=movs.filter(m=>m.movement_type==='in').reduce((s,m)=>s+(m.movement_qty||0)*(m.movement_price||0),0);
  const totalOut=movs.filter(m=>m.movement_type==='out').reduce((s,m)=>s+(m.movement_qty||0)*(m.movement_price||0),0);
  const remaining=prods.reduce((s,p)=>s+(p.quantity||0),0);
  const profit=totalOut-totalIn;
  
  let csv='\uFEFF'+'تقرير حركة المخزون\n';
  csv+='شركة سنك لقطع غيار السيارات\n';
  csv+=`الفترة,من ${from} إلى ${to}\n`;
  csv+=`تم الإنشاء,${new Date().toLocaleString('ar-EG')}\n\n`;
  csv+='ملخص التقرير\n';
  csv+='المؤشر,القيمة\n';
  csv+=`إجمالي الوارد,${fmt(totalIn)} ${cs()}\n`;
  csv+=`إجمالي الصادر,${fmt(totalOut)} ${cs()}\n`;
  csv+=`الكمية المتبقية,${remaining} وحدة\n`;
  csv+=`الربح/الخسارة,${fmt(profit)} ${cs()}\n\n`;
  csv+='التاريخ,النوع,المنتج,الكمية,السعر,الإجمالي\n';
  
  const sorted=movs.sort((a,b)=>new Date(b.date||0)-new Date(a.date||0));
  sorted.forEach(m=>{
    const prod=prods.find(p=>p.code===m.product_code);
    const total=(m.movement_qty||0)*(m.movement_price||0);
    csv+=`${m.date?new Date(m.date).toLocaleDateString('ar-EG'):'-'},${m.movement_type==='in'?'وارد':'صادر'},"${prod?.name||m.product_code}",${m.movement_qty},${m.movement_price},${total}\n`;
  });
  
  try{
    downloadFile(`تقرير_المخزون_${from}_إلى_${to}.csv`,csv,'text/csv');
    showToast('✓ تم تحميل التقرير (CSV)');
  }catch(e){
    console.error('CSV error:',e);
    showToast('⚠️ حدث خطأ أثناء التحميل','error');
  }
}

async function saveWarehouse(){
  const name=document.getElementById('w-name').value.trim();
  if(!name){showToast('الرجاء إدخال اسم المخزن','error');return}
  const existing=getWarehouses();
  if(existing.includes(name)){showToast('المخزن موجود مسبقاً','error');return}
  if(allData.length>=999){showToast('تم الوصول للحد الأقصى','error');return}
  const r=await window.dataSdk.create({type:'warehouse',name,warehouse:'all'});
  if(r.isOk){
    showToast('تم إضافة المخزن');
    hideModal('warehouse-modal');
    await addLog('إضافة مخزن',name);
    document.getElementById('w-name').value='';
    updateWarehouseSelect();
    updateWarehouseSwitcherMenu();
    renderWarehouses();
  }
  else showToast('حدث خطأ','error');
}

function quickReportGenerateCSV(){
  const from=document.getElementById('quick-report-from').value;
  const to=document.getElementById('quick-report-to').value;
  
  if(!from||!to){showToast('الرجاء تحديد التواريخ','error');return}
  
  const movs=getMovements().filter(m=>{const d=new Date(m.date||0);const f=new Date(from);const t=new Date(to);return d>=f&&d<=t});
  const prods=getProducts();
  const invs=getInvoices().filter(i=>{const d=new Date(i.invoice_date||0);const f=new Date(from);const t=new Date(to);return d>=f&&d<=t});
  
  const totalProducts=prods.length;
  const totalStock=prods.reduce((s,p)=>s+(p.quantity||0),0);
  const totalSales=invs.reduce((s,i)=>s+(i.invoice_total||0),0);
  const totalBuy=movs.filter(m=>m.movement_type==='in').reduce((s,m)=>s+(m.movement_qty||0)*(m.movement_price||0),0);
  const totalSell=movs.filter(m=>m.movement_type==='out').reduce((s,m)=>s+(m.movement_qty||0)*(m.movement_price||0),0);
  const profit=totalSell-totalBuy;
  
  let csv='\uFEFF'+'التقرير السريع\n';
  csv+=`الفترة,من ${from} إلى ${to}\n`;
  csv+=`تاريخ الإنشاء,${new Date().toLocaleString('ar-EG')}\n\n`;
  csv+='ملخص التقرير\n';
  csv+='المؤشر,القيمة\n';
  csv+=`إجمالي المنتجات,${totalProducts}\n`;
  csv+=`إجمالي المخزون,${totalStock} وحدة\n`;
  csv+=`إجمالي المبيعات,${fmt(totalSales)} ${cs()}\n`;
  csv+=`الأرباح,${fmt(profit)} ${cs()}\n`;
  csv+=`إجمالي الوارد,${fmt(totalBuy)} ${cs()}\n`;
  csv+=`إجمالي الصادر,${fmt(totalSell)} ${cs()}\n`;
  
  try{
    downloadFile(`تقرير_سريع_${from}_إلى_${to}.csv`,csv,'text/csv');
    showToast('✓ تم تحميل التقرير (CSV)');
    document.getElementById('quick-report-modal')?.remove();
  }catch(e){
    console.error('CSV error:',e);
    showToast('⚠️ حدث خطأ أثناء التحميل','error');
  }
}

function autoCreateBackup(){
  // Create comprehensive backup with all data
  let backupData='\uFEFF\n=== نسخة احتياطية تلقائية ===\n';
  backupData+=`تاريخ النسخ: ${new Date().toLocaleString('ar-EG')}\n`;
  backupData+=`إجمالي السجلات: ${allData.length}\n\n`;
  
  // Products
  const prods=getProducts();
  if(prods.length>0){
    backupData+='--- المنتجات ---\n';
    backupData+='الكود,الاسم,سعر الشراء,سعر البيع,الكمية\n';
    prods.forEach(p=>{backupData+=`${p.code},${p.name},${p.buy_price},${p.sell_price},${p.quantity}\n`});
    backupData+='\n';
  }
  
  // Movements
  const movs=getMovements();
  if(movs.length>0){
    backupData+='--- حركات المخزون ---\n';
    backupData+='التاريخ,النوع,المنتج,الكمية,السعر\n';
    movs.forEach(m=>{backupData+=`${m.date?new Date(m.date).toLocaleDateString('ar-EG'):'-'},${m.movement_type},${m.product_code},${m.movement_qty},${m.movement_price}\n`});
    backupData+='\n';
  }
  
  // Invoices
  const invs=getInvoices();
  if(invs.length>0){
    backupData+='--- الفواتير ---\n';
    backupData+='الرقم,العميل,التاريخ,الإجمالي\n';
    invs.forEach(i=>{backupData+=`${i.invoice_number},${i.customer_name},${i.invoice_date?new Date(i.invoice_date).toLocaleDateString('ar-EG'):'-'},${i.invoice_total}\n`});
    backupData+='\n';
  }
  
  downloadFile(`نسخة_احتياطية_شاملة_${new Date().toISOString().split('T')[0]}.txt`,backupData,'text/plain');
  showToast('✓ تم حفظ النسخة الاحتياطية تلقائياً!');
}

function checkAndAutoBackup(){
  const recordCount=allData.length;
  if(recordCount>=900&&!window.backupAlerted){
    window.backupAlerted=true;
    showToast('⚠️ اقتربت من الحد الأقصى! جاري حفظ نسخة احتياطية تلقائية...',undefined);
    setTimeout(()=>{autoCreateBackup()},1000);
  }
}
